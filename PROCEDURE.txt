CREATE OR REPLACE PROCEDURE EMP_DD(N IN VASU.EMPNO%TYPE) IS
NA VASU.ENAME%TYPE;
SA VASU.SAL%TYPE;
BEGIN
SELECT ENAME,SAL INTO NA,SA FROM VASU WHERE EMPNO=N;
DBMS_OUTPUT.PUT_LINE('EMPLOYEE DETAILS');
DBMS_OUTPUT.PUT_LINE('______________________');
DBMS_OUTPUT.PUT_LINE(N|| ' ' ||NA ||' ' ||SA);
END;
